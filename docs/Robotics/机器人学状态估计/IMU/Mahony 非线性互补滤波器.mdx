import numpy as np

# 1. 核心调参参数
Kp = 2.0      # 比例增益：控制加速度计纠正陀螺仪漂移的速度（越大收敛越快，但怕震动）
Ki = 0.005    # 积分增益：消除陀螺仪的固有零点偏置（常置一个极小的值，不需要时可设为 0）

# 2. 内部全局变量（用于积分项累加误差）
e_integral_x = 0.0
e_integral_y = 0.0
e_integral_z = 0.0

def mahony_update_6dof(q, gx, gy, gz, ax, ay, az, dt):
    """
    q: 当前姿态四元数 [qw, qx, qy, qz]
    gx, gy, gz: 陀螺仪角速度 (单位: rad/s)
    ax, ay, az: 加速度计数据 (任意单位，只要是三轴即可)
    dt: 采样时间间隔 (单位: 秒)
    """
    global e_integral_x, e_integral_y, e_integral_z
    qw, qx, qy, qz = q

    # 【第一步】归一化加速度计数据
    norm = np.sqrt(ax**2 + ay**2 + az**2)
    if norm == 0.0: return q  # 处理异常
    ax, ay, az = ax / norm, ay / norm, az / norm

    # 【第二步】提取当前四元数姿态下的“理论重力方向”
    # 这是四元数旋转矩阵的第三列转置，代表地理坐标系下的 [0, 0, 1]（重力）转到机体坐标系的值
    vx = 2.0 * (qx * qz - qw * qy)
    vy = 2.0 * (qw * qx + qy * qz)
    vz = qw**2 - qx**2 - qy**2 + qz**2

    # 【第三步】通过向量叉乘（Cross Product）计算误差
    # 实际重力 [ax, ay, az] 与 理论重力 [vx, vy, vz] 的叉乘，反映了姿态的偏差方向和大小
    ex = (ay * vz - az * vy)
    ey = (az * vx - ax * vz)
    ez = (ax * vy - ay * vx)

    # 【第四步】PI 控制器：对误差进行比例和积分
    if Ki > 0.0:
        e_integral_x += ex * dt
        e_integral_y += ey * dt
        e_integral_z += ez * dt
    else:
        e_integral_x, e_integral_y, e_integral_z = 0.0, 0.0, 0.0

    # 将误差补偿应用到陀螺仪数据上
    gx += Kp * ex + Ki * e_integral_x
    gy += Kp * ey + Ki * e_integral_y
    gz += Kp * ez + Ki * e_integral_z

    # 【第五步】用补偿后的角速度更新四元数微分方程
    q_dot_w = 0.5 * (-qx * gx - qy * gy - qz * gz)
    q_dot_x = 0.5 * ( qw * gx + qy * gz - qz * gy)
    q_dot_y = 0.5 * ( qw * gy - qx * gz + qz * gx)
    q_dot_z = 0.5 * ( qw * gz + qx * gy - qy * gx)

    # 积分得到新的四元数
    qw += q_dot_w * dt
    qx += q_dot_x * dt
    qy += q_dot_y * dt
    qz += q_dot_z * dt

    # 【第六步】归一化四元数并输出
    norm = np.sqrt(qw**2 + qx**2 + qy**2 + qz**2)
    return [qw / norm, qx / norm, qy / norm, qz / norm]
